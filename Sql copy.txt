SELECT DAL.XXASU_DECRYPT_RESTWS_CALL(SHPORD.RECEIVER_NAME) FULL_NAME
     , DAL.XXASU_DECRYPT_RESTWS_CALL(NVL(SHPORD.BUSINESS_NAME, SHPADR.BUSINESS_NAME)) BUSINESS_NAME
     , DAL.XXASU_DECRYPT_RESTWS_CALL(SHPADR.ADDRESS_LINE_1) ADDRESS_LINE_1
     , DAL.XXASU_DECRYPT_RESTWS_CALL(SHPADR.ADDRESS_LINE_2) ADDRESS_LINE_2
     , SHPADR.CITY_NAME
     , SHPADR.STATE_PROVINCE_CODE
     , SHPADR.POSTAL_CODE POSTAL_CODE
     , SHPADR.COUNTRY_CODE
     , AST.SUBSCRIPTION_NBR
     , DAL.XXASU_DECRYPT_RESTWS_CALL(AST.MOBILE_DEVICE_NBR) MOBILE_DEVICE_NBR
     , SRVORD.SERVICE_ORDER_ID
     , SRVORD.SERVICE_ORDER_NBR
     , SRVORD.PARENT_SERVICE_ORDER_ID
     , SRVORD.CREATED_DATE,
     TRUNC(SRVORD.CREATED_DATE) 
     , CSTCAS.CUSTOMER_CASE_NBR
     , SRVRST.SERVICE_REQUEST_NBR
     , CLNACT.CLIENT_CHANNEL_ID
  FROM CUSTOMER.SERVICE_ORDER SRVORD
 INNER JOIN CUSTOMER.SHIPPING_ORDER SHPORD ON SHPORD.SERVICE_ORDER_ID = SRVORD.SERVICE_ORDER_ID AND SHPORD.ACTIVE_IND = 'Y'
 INNER JOIN REFERENCE.ADDRESS SHPADR ON SHPADR.ADDRESS_ID = SHPORD.RECEIVER_ADDRESS_ID AND SHPADR.ACTIVE_IND = 'Y'
 INNER JOIN ASSET.ASSET AST ON AST.ASSET_ID = SRVORD.ASSET_ID AND AST.ACTIVE_IND = 'Y'
 INNER JOIN CUSTOMER.CUSTOMER_CASE_CATEGORY CSTCASC ON CSTCASC.ITEM_TYPE_CODE = 'SVCORD' AND CSTCASC.ITEM_ID = SRVORD.SERVICE_ORDER_ID AND CSTCASC.ACTIVE_IND = 'Y'
 INNER JOIN CUSTOMER.CUSTOMER_CASE CSTCAS ON CSTCAS.CUSTOMER_CASE_ID = CSTCASC.CUSTOMER_CASE_ID AND CSTCAS.ACTIVE_IND = 'Y'
 INNER JOIN CUSTOMER.SERVICE_REQUEST SRVRST ON SRVRST.SERVICE_REQUEST_ID = CSTCASC.SERVICE_REQUEST_ID AND SRVRST.ACTIVE_IND = 'Y'
 INNER JOIN CLIENT.CLIENT_ACCOUNT CLNACT ON AST.CLIENT_ACCOUNT_ID = CLNACT.CLIENT_ACCOUNT_ID
 INNER JOIN CLIENT.CLIENT_CHANNEL CLNCNL ON CLNCNL.CLIENT_CHANNEL_ID = CLNACT.CLIENT_CHANNEL_ID
 WHERE 1=1 
   AND SRVORD.SERVICE_ORDER_TYPE_CODE = 'ENVLPONLY'
   AND SRVORD.SERVICE_ORDER_STATUS_CODE IN ('READY','SHIPPED')
   AND SRVRST.FULFILLMENT_OPTION_CODE = 'RPLCMNT'
  -- AND TRUNC(SRVORD.CREATED_DATE) BETWEEN TRUNC(SYSDATE-1) AND TRUNC(SYSDATE-1)
   AND TRUNC(SRVORD.CREATED_DATE) = TRUNC(SYSDATE-1)
   --AND SRVORD.VENDOR_ID = HEXTORAW('56587B59E4F10893E0538A1F080A6D37') -- ATT VENDOR_ID Hard coded
   AND CLNCNL.CLIENT_CHANNEL_ID = HEXTORAW('4ECA222A37B63627E0536138060AB356') -- ATT CLIENT_CHANNEL_ID
   --AND SRVORD.SERVICE_ORDER_ID = HEXTORAW('1E878A28C64501E0A0AA11D71FA01E8F')
   --AND CSTCAS.CUSTOMER_CASE_NBR IN ('901444717244','906919629757')
   ;
